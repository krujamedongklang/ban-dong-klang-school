# คู่มือการตั้งค่า Supabase (Supabase Configuration Guide)

คู่มือนี้แนะนำวิธีการเตรียมตารางข้อมูลและตั้งค่าความปลอดภัย (Row Level Security - RLS) ในระบบ Supabase ของคุณเพื่อนำมาใช้งานร่วมกับเว็บไซต์โรงเรียนบ้านดงกลาง (ธนาคารกรุงเทพ ๓๑)

---

## ขั้นตอนที่ 1: การสร้างตารางข้อมูลและนโยบายความปลอดภัย (SQL Editor)

ให้คุณเข้าไปที่ **Supabase Dashboard** ของคุณ เลือกโปรเจกต์ที่ต้องการ จากนั้นไปที่เมนู **SQL Editor** ด้านซ้ายมือ กดปุ่ม **New Query** แล้วคัดลอกคำสั่ง SQL ด้านล่างนี้ไปวางและกดปุ่ม **Run** เพื่อสร้างฐานข้อมูล:

```sql
-- ==========================================================================
-- 1. สร้างตารางข่าวสาร (news)
-- ==========================================================================
CREATE TABLE IF NOT EXISTS public.news (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    title TEXT NOT NULL,
    content TEXT NOT NULL,
    date DATE NOT NULL DEFAULT CURRENT_DATE,
    category TEXT NOT NULL CHECK (category IN ('activity', 'academic')),
    image_url TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) NOT NULL
);

-- ==========================================================================
-- 2. สร้างตารางการติดต่อ (contact_submissions)
-- ==========================================================================
CREATE TABLE IF NOT EXISTS public.contact_submissions (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT NOT NULL,
    email TEXT NOT NULL,
    subject TEXT NOT NULL,
    message TEXT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) NOT NULL
);

-- ==========================================================================
-- 3. เปิดการใช้งานระบบ Row Level Security (RLS) เพื่อความปลอดภัย
-- ==========================================================================
ALTER TABLE public.news ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.contact_submissions ENABLE ROW LEVEL SECURITY;

-- ==========================================================================
-- 4. ตั้งค่านโยบายความปลอดภัยสำหรับตารางข่าวสาร (news Policies)
-- ==========================================================================
-- อนุญาตให้บุคคลทั่วไปดึงข้อมูลข่าวสารไปอ่านได้ (SELECT)
CREATE POLICY "Allow public read access to news" 
ON public.news 
FOR SELECT 
TO public 
USING (true);

-- อนุญาตเฉพาะผู้ใช้ที่ผ่านการเข้าสู่ระบบ (แอดมิน) ในการเพิ่ม/แก้ไข/ลบข่าวสาร (INSERT/UPDATE/DELETE)
-- *หมายเหตุ: คุณสามารถเข้าไปจัดการข้อมูลผ่านระบบ Supabase Table Editor ได้โดยตรงโดยไม่ต้องใช้นโยบายนี้
CREATE POLICY "Allow authenticated insert to news" 
ON public.news 
FOR INSERT 
TO authenticated 
WITH CHECK (true);

CREATE POLICY "Allow authenticated update to news" 
ON public.news 
FOR UPDATE 
TO authenticated 
USING (true) 
WITH CHECK (true);

CREATE POLICY "Allow authenticated delete to news" 
ON public.news 
FOR DELETE 
TO authenticated 
USING (true);

-- ==========================================================================
-- 5. ตั้งค่านโยบายความปลอดภัยสำหรับตารางติดต่อสอบถาม (contact_submissions Policies)
-- ==========================================================================
-- อนุญาตให้บุคคลทั่วไปสามารถเขียนหรือกดส่งฟอร์มข้อความติดต่อสอบถามเข้ามาได้ (INSERT)
CREATE POLICY "Allow public insert to contact_submissions" 
ON public.contact_submissions 
FOR INSERT 
TO public 
WITH CHECK (true);

-- *ไม่มีนโยบายการเปิดให้บุคคลทั่วไปอ่านข้อมูลข่าวสารการติดต่อ (SELECT) 
-- ส่งผลให้ข้อมูลติดต่อของผู้ใช้ที่ส่งเข้ามาเป็นความลับขั้นสุด มีเพียงคุณที่เป็นเจ้าของฐานข้อมูลเท่านั้นที่จะดูได้ผ่าน Dashboard
```

## ขั้นตอนที่ 2: การดึงคีย์ API และตั้งค่าบนเว็บไซต์

1. ไปที่เมนู **Project Settings** (ไอคอนฟันเฟืองที่มุมซ้ายล่างของ Supabase Dashboard)
2. เลือกเมนูย่อย **API**
3. คัดลอกค่าสองส่วนนี้:
   - **Project URL** (ในช่อง URL)
   - **anon / public** (ในช่อง API Keys)
4. เปิดไฟล์ `.env` ที่อยู่ในโฟลเดอร์เว็บไซต์ของคุณ
5. แก้ไขบรรทัดต่อไปนี้ด้วยค่าที่คุณคัดลอกมา:
   ```env
   VITE_SUPABASE_URL=https://[ไอดีโปรเจกต์ของคุณ].supabase.co
   VITE_SUPABASE_ANON_KEY=[คีย์ anon/public ของคุณ]
   ```
6. บันทึกไฟล์ให้เรียบร้อย เท่านี้ระบบดึงข่าวสารและระบบส่งข้อความจะเชื่อมต่อกับ Supabase จริงทันที!

## ขั้นตอนที่ 3: เพิ่มข้อมูลข่าวสารตัวอย่าง (ทำผ่าน Supabase Table Editor)

เพื่อทดสอบระบบดึงข่าวสารของหน้าเว็บ คุณสามารถคลิกเข้าไปที่เมนู **Table Editor** ของ Supabase เลือกตาราง `news` จากนั้นกดปุ่ม **Insert row** เพื่อเพิ่มข่าวสารตัวอย่าง เช่น:
- **title**: กิจกรรมต้อนรับเปิดเทอมใหม่ โรงเรียนบ้านดงกลาง
- **content**: `<p>เนื้อหารายละเอียดของข่าวสาร...</p>` (แนะนำให้เขียนโดยใช้ HTML Tag เช่น `<p>` หรือ `<b>` เพื่อการแสดงผลที่สวยงาม)
- **date**: `2026-06-30`
- **category**: `activity` (สามารถเลือกใส่ได้ระหว่าง `activity` หรือ `academic`)
- **image_url**: ใส่ลิงก์รูปภาพของคุณ (หรือเว้นว่างไว้เพื่อแสดงผลรูปโลโก้โรงเรียนแบบออฟไลน์)
